    Supabase のテーブル定義と RLS の例はページ下部に記載しています

    Supabase テーブル例と RLS（コピーして SQL 実行）

    -- quizzes テーブル
create table quizzes (
  id uuid primary key default gen_random_uuid(),
  title text not null,
  genre text not null,
  created_at timestamptz default now()
);

-- questions テーブル
create table questions (
  id uuid primary key default gen_random_uuid(),
  quiz_id uuid references quizzes(id) on delete cascade,
  text text not null,
  choices jsonb not null, -- ["選択肢A","選択肢B",...]
  answer_index int not null,
  created_at timestamptz default now()
);

-- results テーブル
create table results (
  id uuid primary key default gen_random_uuid(),
  quiz_id uuid references quizzes(id),
  score int,
  total int,
  created_at timestamptz default now()
);

-- RLS の簡易例（公開読み取り、書き込みは制限）
-- quizzes
alter table quizzes enable row level security;
create policy "public_select" on quizzes for select using (true);

-- questions
alter table questions enable row level security;
create policy "public_select" on questions for select using (true);

-- results
alter table results enable row level security;
create policy "insert_results" on results for insert using (true) with check (true);
create policy "select_results" on results for select using (true);
    注：学習用途であれば上の簡易ポリシーで始められますが、公開書き込みを制限したい場合は認証を使ってポリシーを厳しくしてください
